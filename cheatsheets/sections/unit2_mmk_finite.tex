\section*{Multiserver systems; finite buffer}
\textbf{M/M/$k$:} common queue, i.i.d.\ exp.\ servers.\\
Metric: queueing prob.\ $P_Q=\Prob(\text{wait before service})$.\\
$\mu=1/E[S]$;\ $C=\lambda/\mu$;\ $\rho=\lambda/(k\mu)=C/k$ (stable $\rho<1$).\\
$p_0=\Bigl[\sum_{j=0}^{k-1}\frac{C^j}{j!}+\frac{C^k}{k!(1-\rho)}\Bigr]^{-1}$;\quad
$P_Q=p_0\,\dfrac{C^k}{k!(1-\rho)}$.\\
\textbf{Steps:} try $k=1,2,\ldots$ until $P_Q\le$ target; min $k$.\\
\textbf{Trap:} identical servers assumed for this $C$.

\textbf{M/M/1/$U$:} max $U$ jobs in system; buffer $U-1$.\\
$\pi_n=(\lambda/\mu)^n\pi_0$;\quad
$\pi_0=\dfrac{1-\lambda/\mu}{1-(\lambda/\mu)^{U+1}}$.\\
Blocking $=\pi_U$;\quad $X=\lambda(1-\pi_U)$.\\
$P(\text{wait})=\sum_{n=1}^{U-1}\pi_n=1-\pi_0-\pi_U$.\\
\textbf{Steps (b):} increase $U$ until $P(\text{wait})\le$ target (less blocking, higher $X$).
