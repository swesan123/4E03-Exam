\section*{6.\ Continuous-time Markov chains}
\textbf{Idea.}\ Transitions at any continuous time;\ countable states;\ Markov + stationary (homogeneous).\\
\textbf{Residence time in state.}\ Exponential (competing exponential clocks).\\
\textbf{Limiting probs $\pi$.}\ Balance: rate into state $=$ rate out (GBE);\ or local balance (birth--death).\\
\textbf{Birth--death.}\ General $\pi_n$ product form $\pi_n=\pi_0\prod_{i<n}(\lambda_i/\mu_{i+1})$.\\
\textbf{$M/M/1$ embedded.}\ $p_n=(1-\rho)\rho^n$,\ $\rho=\lambda/\mu<1$;\ busy prob.\ $\rho$;\ $\E[N]=\rho/(1-\rho)$;\ Little: $\E[T]=\E[N]/\lambda$.\\
\textbf{$M/M/1/U$ (finite).}\ $\pi_n=(\lambda/\mu)^n\pi_0$;\ $\pi_0=\dfrac{1-\lambda/\mu}{1-(\lambda/\mu)^{U+1}}$;\ blocking $\pi_U$;\ $X=\lambda(1-\pi_U)$.\\
\textbf{$M/M/k$.}\ See topic 8 (Erlang-$C$ / $P_Q$).\\
\textbf{Client/server CTMC.}\ State $=$ jobs at server;\ think exp.$(\zeta)$,\ service exp.$(\mu)$;\ policy often irrelevant for CTMC size.
