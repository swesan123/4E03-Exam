\section*{1.\ Probability introduction}
\textbf{Events / probability.}\ $0\le P(E)\le1$;\quad $P(E\cup F)=P(E)+P(F)-P(E\cap F)$.\\
\textbf{Conditional.}\ $P(E\mid F)=P(E\cap F)/P(F)$.\\
\textbf{Expectation / variance.}\ $\E[X]=\sum x\,p(x)$ or $\int x f(x)\,dx$;\quad $\Var(X)=\E[X^2]-(\E[X])^2$;\quad $\E[X^2]\ge\Var(X)$.\\
\textbf{Discrete template.}\ Given pmf: normalize $\sum p_i=1$; $\E[g(X)]=\sum g(x_i)p_i$.\\
\textbf{Uniform $[a,b]$.}\ $\Var=(b-a)^2/12$.
