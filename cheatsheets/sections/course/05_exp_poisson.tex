\section*{5.\ Exponential distribution \& Poisson process}
\textbf{Exp$(\lambda)$ memoryless.}\ $P(X>s+t\mid X>s)=P(X>t)$.\\
\textbf{Min of independ.\ exp.}\ $\min(X_1,X_2)\sim\mathrm{Exp}(\lambda_1+\lambda_2)$.\\
\textbf{Counting process $N(t)$.}\ Nondecreasing;\ $N(t)\in\{0,1,\ldots\}$.\\
\textbf{Poisson rate $\lambda$.}\ Indep.\ increments;\ $N(t+s)-N(s)\sim\mathrm{Pois}(\lambda t)$;\ equiv.\ i.i.d.\ $\mathrm{Exp}(\lambda)$ inter-arrivals.\\
\textbf{Merge.}\ Indep.\ Poisson$(\lambda_1)$, Poisson$(\lambda_2)$ $\Rightarrow$ sum rate $\lambda_1+\lambda_2$.\\
\textbf{Split.}\ Each event type A w.p.\ $p$ $\Rightarrow$ Poisson$(p\lambda)$ and $(1-p)\lambda$, indep.
